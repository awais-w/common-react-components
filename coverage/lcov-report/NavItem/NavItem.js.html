import React from 'react';
import { CSSTransition } from 'react-transition-group';
import PropTypes from 'prop-types';
import './NavItem.scss';
 
const NavItem = ({ title, subTitle, summary, isActive, onClick, isComplete, currIndex }) => (
  <div
    onClick={() => onClick({ desiredIndex: currIndex })}
    className={`step-title-container ${isActive ? 'active-item' : ''}`}>
    <div className='step-title-container__title'>
      {isComplete ? (
        <CSSTransition in timeout={300} classNames='step-title-container__complete' unmountOnExit>
          <div className='step-title-container__complete'>â˜‘</div>
        </CSSTransition>
      ) : (
        ''
      )}
      {title}
    </div>
 
    {/* Only show description of step for active one */}
    {isActive && <div className='step-title-container__subtitle'>{subTitle}</div>}
 
    {/* Show summaries on previous steps if we've got one */}
    {isComplete && <div className='step-title-container__summary'>{summary}</div>}
  </div>
);
 
NavItem.propTypes = {
  /**
   * Primary title for this step
   */
  title: PropTypes.string.isRequired,
  /**
   * Called when we click anywhere in nav item
   */
  onClick: PropTypes.func,
  /**
   * Description about whats required from the step
   */
  subTitle: PropTypes.string,
  /**
   * Show whats been completed in a previous step
   */
  summary: PropTypes.oneOfType([PropTypes.string, PropTypes.element]),
  /**
   * Whether we're currently displaying the content for this step
   */
  isActive: PropTypes.bool.isRequired,
  /**
   * Has this stage already been completed
   */
  isComplete: PropTypes.bool,
  /**
   * The current index we've reached upto
   */
  currIndex: PropTypes.number.isRequired,
};
 
NavItem.defaultProps = {
  onClick: null,
  summary: '',
  subTitle: '',
  isComplete: true,
};
 
export default NavItem;
