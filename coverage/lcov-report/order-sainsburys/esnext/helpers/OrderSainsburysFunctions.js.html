export const sortPickedLines = (pickedLinesArr) => {
  const pickedLines = _.cloneDeep(pickedLinesArr);
 
  return pickedLines.sort((aa, bb) => {
    // Sort by return status
    const isNonDeliveredItemA = Boolean(aa.returnReason);
    const isNonDeliveredItemB = Boolean(bb.returnReason);
    if (isNonDeliveredItemA || isNonDeliveredItemB) {
      // -1 a goes first
      // 0 neither goes first
      // 1 b goes first
      return isNonDeliveredItemB - isNonDeliveredItemA;
    }
 
    // If none, don't sort
    return 0;
  });
};
 
export const sortOrderAndPickedLines = (lineItemsArr) => {
