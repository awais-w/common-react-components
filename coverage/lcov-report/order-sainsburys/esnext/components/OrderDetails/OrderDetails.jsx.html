import React, { Component, Fragment } from 'react';
import bem from '@argos/utils/esnext/bem';
import moment from 'moment';
import Pane from '@cmc/pane/esnext/pane';
import './order-details.scss';
import IconLabelValue from '@cmc/icon-label-value/esnext/IconLabelValue';
import OrderDetailsBody from '../OrderDetailsBody/OrderDetailsBody';
import propTypes from './OrderDetails.propTypes';
 
const cn = bem({ prefix: 'cmc-', block: 'order-details-sainsburys' });
class OrderDetails extends Component {
  render() {
    const {
      id,
      status,
      statusUpdateDateTime,
      bagless,
      largePrint,
      lineItems,
      loyaltyCards,
      isLoading,
      deliveryTracking,
    } = this.props;
 
    const additionalInfo = () => {
      const additional = (
        <div className={cn('additional')}>
          {bagless && <IconLabelValue icon={'check_circle'} label='Bagless' />}
          {largePrint && <IconLabelValue icon={'check_circle'} label='Large Print' />}
        </div>
      );
      return bagless || largePrint ? additional : null;
    };
 
    return (
      <Fragment>
        <Pane
          className={cn()}
          title1='Order Details'
          title2={!isLoading && `Order ID: ${id}`}
          title3={!isLoading && `${moment(statusUpdateDateTime).format('HH:mm DD/MM/YY')}: ${status}`}
          icon='receipt'
          isExpanded={true}
          titleRight={additionalInfo()}
          children={
            <OrderDetailsBody
              lineItems={lineItems}
              loyaltyCards={loyaltyCards}
              isLoading={isLoading}
              deliveryTracking={deliveryTracking}
            />
          }
        />
      </Fragment>
    );
  }
}
OrderDetails.propTypes = propTypes;
OrderDetails.defaultProps = {};
export default OrderDetails;
