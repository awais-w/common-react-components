import React, { Fragment, useState } from 'react';
import bem from '@argos/utils/esnext/bem';
import isEmpty from '@cmc/utils/esnext/isEmpty';
import CreditCardDetails from '@cmc/credit-card-details/esnext/CreditCardDetails';
import './order-details-body.scss';
import Card from '@cmc/card/esnext/card';
import DeliveryTracking from '@cmc/delivery-details/esnext/components/delivery-tracking';
import LineItem from '../LineItem/LineItem';
import propTypes from './OrderDetailsBody.propTypes';
 
const cn = bem({ prefix: 'cmc-', block: 'order-details-sainsburys' });
 
const OrderDetailsBody = (props) => {
  const { lineItems, loyaltyCards, isLoading, deliveryTracking } = props;
  const [isPanelExpanded, setIsPanelExpanded] = useState(true);
  const lineItemsArr = isLoading ? (
    <Card isLoading />
  ) : (
    lineItems.map((lineItem) => <LineItem data={lineItem} key={lineItem.orderedLine.ean} />)
  );
 
  const loyaltyCardsArr = !isEmpty(loyaltyCards)
    ? loyaltyCards.map((loyaltyCard) => (
        <CreditCardDetails
          type={loyaltyCard.loyaltyType}
          number={loyaltyCard.id}
          startDate={loyaltyCard.validFrom}
          expiryDate={loyaltyCard.validTo}
          name={loyaltyCard.nameOnCard}
          isCompact={true}
          key={loyaltyCard.id}
        />
      ))
    : [];
 
  const uberDetails = !isEmpty(deliveryTracking) ? (
    <div className={cn('highlight')}>
      <Card
        isExpanded={isPanelExpanded}
        title1='7734458'
        title2='UBER Delivery (Click & Collect)'
        panelType='panel-notify'
        onToggle={() => setIsPanelExpanded(!isPanelExpanded)}>
        <div className={cn('delivery-tracking')}>
          <DeliveryTracking
            actualArrival={deliveryTracking.actualArrival}
            plannedArrival={deliveryTracking.plannedArrival}
            dropNumber={deliveryTracking.dropNumber}
            bookedSlot={deliveryTracking.bookedSlot}
            price={deliveryTracking.price}
            origin={deliveryTracking.origin}
            instructions={deliveryTracking.instructions}
          />
        </div>
      </Card>
    </div>
  ) : null;
 
  return (
    <Fragment>
      <div className={cn('body')}>
        <div className={cn('loyalty-cards')}>{loyaltyCardsArr}</div>
        {uberDetails}
        <div className={cn('order-lines')}>{lineItemsArr}</div>
      </div>
    </Fragment>
  );
};
 
OrderDetailsBody.propTypes = propTypes;
export default OrderDetailsBody;
