import PropTypes from 'prop-types';
 
const propTypes = {
  orderNo: PropTypes.string,
  shouldRequestUseCurrentHost: PropTypes.bool,
  isLoading: PropTypes.bool,
  errorMessage: PropTypes.string,
  data: PropTypes.shape({
    id: PropTypes.string,
    status: PropTypes.string,
    bagless: PropTypes.bool,
    largePrint: PropTypes.bool,
    lineItems: PropTypes.arrayOf(
      PropTypes.shape({
        status: PropTypes.object,
        orderedLine: PropTypes.object,
        pickedLine: PropTypes.object,
        pickedResult: PropTypes.string,
      }),
    ),
    loyaltyCards: PropTypes.arrayOf(
      PropTypes.shape({
        id: PropTypes.string,
        loyaltyType: PropTypes.string,
        validFrom: PropTypes.string,
        validTo: PropTypes.string,
        nameOnCard: PropTypes.string,
      }),
    ),
    // TODO: Is this the order date?
    statusUpdateDateTime: PropTypes.string,
    deliveryAmount: {
      amount: PropTypes.number,
      currency: PropTypes.string,
    },
    deliveries: PropTypes.arrayOf(
      PropTypes.shape({
        details: PropTypes.shape({
          address: PropTypes.shape({}),
          instructions: PropTypes.string,
        }),
        contact: PropTypes.shape({
          name: {
            middle: PropTypes.string,
            first: PropTypes.string,
            last: PropTypes.string,
            business: PropTypes.string,
            title: PropTypes.string,
          },
          contactDetails: {
            emailAddress: PropTypes.arrayOf(
              PropTypes.shape({
                type: PropTypes.string,
                address: PropTypes.string,
              }),
            ),
            telephoneNumbers: PropTypes.arrayOf(
              PropTypes.shape({
                type: PropTypes.string,
                number: PropTypes.string,
              }),
            ),
          },
        }),
        run: PropTypes.object,
        slot: PropTypes.shape({
          displayText: PropTypes.string,
          storeId: PropTypes.string,
        }),
        clickAndCollect: PropTypes.bool,
      }),
    ),
    payment: PropTypes.shape({
      token: PropTypes.string,
      eci: PropTypes.string,
      transactionStatus: PropTypes.string,
      xid: PropTypes.string,
      dsTransactionId: PropTypes.string,
      threeDsVersionNumber: PropTypes.string,
    }),
    colleagueDiscountCard: PropTypes.string,
    refunded: PropTypes.bool,
    vouchers: PropTypes.array,
    totalAmount: PropTypes.object.isRequired,
    orderTransactions: PropTypes.array.isRequired,
    shipNode: PropTypes.shape({
      id: PropTypes.number.isRequired,
      name: PropTypes.string.name.isRequired,
    }),
    canRetryPayment: PropTypes.bool,
  }),
  onRetryPayment: PropTypes.func,
  retryPaymentStatus: PropTypes.string,
  userHasRetryPermissions: PropTypes.bool,
};
 
export default propTypes;
