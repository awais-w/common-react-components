import React from 'react';
import bem from '@argos/utils/esnext/bem';
import formatCurrency from '@cmc/utils/esnext/formatCurrency';
import isEmpty from '@cmc/utils/esnext/isEmpty';
import IconLabelValue from '@cmc/icon-label-value/esnext';
import './picked-line-header.scss';
 
const cn = bem({ prefix: 'cmc-', block: 'picked-line-header' });
 
export default function PickedLineHeader({ data }) {
  const { description = 'N/A', price = {}, quantity, returnReason, sku, substitute, unitOfMeasure, shortLife } = data;
 
  return (
    <div className={cn('title-wrapper')}>
      <span className={cn('information')}>
        {sku && <span className={cn('sku')}>{sku} - </span>}
        {description}
      </span>
      {quantity && !isEmpty(price) && (
        <span className={cn('price')}>
          {`${quantity} x ${price.currency &&
            formatCurrency({
              input: price.currency,
              outputFormat: 'symbol',
            })}`}
          {price.amount} {unitOfMeasure && `/ ${unitOfMeasure.toUpperCase()}`}
        </span>
      )}
      <span className={cn('icons')}>
        {shortLife && (
          <IconLabelValue
            className={cn('short-life-icon')}
            iconSource={'google'}
            icon='timer'
            tooltipValue='Short life'
          />
        )}
        {!isEmpty(returnReason) && (
          <IconLabelValue className={cn('return-icon')} iconSource={'google'} icon='warning' tooltipValue='Return' />
        )}
        {substitute && (
          <IconLabelValue
            className={cn('sub-icon')}
            iconSource={'google'}
            icon='swap_horiz'
            tooltipValue='Substitute'
          />
        )}
      </span>
    </div>
  );
}
