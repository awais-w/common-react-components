import React from 'react';
import bem from '@argos/utils/esnext/bem';
import IconLabelValue from '@cmc/icon-label-value/esnext';
import './picked-line-body.scss';
 
const cn = bem({ prefix: 'cmc-', block: 'picked-line-body' });
 
export default function PickedLineBody({ data }) {
  const { ean = 'N/A', orderType = 'N/A', returnReason, useByDate, shortLife } = data;
 
  return (
    <div className={cn('body-wrapper')}>
      {useByDate && (
        <div className={cn('use-by-date')}>
          <IconLabelValue icon={shortLife ? 'timer' : undefined} label='Use by date' value={useByDate} />
        </div>
      )}
      {returnReason && (
        <div className={cn('return-reason')}>
          <IconLabelValue icon='warning' label='Return reason' value={returnReason} />
        </div>
      )}
      <div className={cn('ean')}>
        <IconLabelValue label='EAN' value={ean} />
      </div>
      <div className={cn('order-type')}>
        <IconLabelValue label='Order Type' value={orderType || 'N/A'} />
      </div>
    </div>
  );
}
