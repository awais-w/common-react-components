import React from 'react';
import PropTypes from 'prop-types';
import bem from '@argos/utils/esnext/bem';
import formatMoney from '@cmc/utils/esnext/formatMoney';
import IconLabelValue from '@cmc/icon-label-value/esnext/IconLabelValue';
import './line-item-header.scss';
 
const cn = bem({ prefix: 'cmc-', block: 'line-item-header' });
 
export default function LineItemHeader({ data }) {
  const {
    pickedResult,
    orderedLine: { sku, description, actualUnitPrice, regularUnitPrice, orderedQuantity, unitOfMeasure },
    pickedLines,
  } = data;
 
  const isEverythingDelivered = pickedLines.every((pickedLine) => !pickedLine.returnReason);
  const isDifferentPrice = regularUnitPrice.amount !== actualUnitPrice.amount;
  const isAnythingSubbed = pickedResult === 'SUBSTITUTION' || pickedResult === 'PART_SUBSTITUTION';
  const pickedQuantity = pickedLines.reduce((acc, pickedLine) => {
    acc += pickedLine.quantity;
    return acc;
  }, 0);
 
  return (
    <div className={cn('title-wrapper')}>
      <span className={cn('information')}>
        <span className={cn('sku')}>{sku}</span> - {description}
      </span>
      <span className={cn('quantity-with-price')}>
        <span className={cn('quantity')}>
          {orderedQuantity} {pickedQuantity !== orderedQuantity && `(${pickedQuantity} picked)`}
        </span>{' '}
        x{' '}
        <span className={cn('pricing')}>
          {isDifferentPrice && (
            <span className={cn('orderline-regular-price')}>
              {formatMoney(regularUnitPrice.amount, regularUnitPrice.currency)}
            </span>
          )}
          {formatMoney(actualUnitPrice.amount, actualUnitPrice.currency)} / {unitOfMeasure.toUpperCase()}
        </span>
      </span>
      <span className={cn('icons')}>
        {!isEverythingDelivered && (
          <IconLabelValue className={cn('return-icon')} icon='warning' tooltipValue='Return' />
        )}
        {isAnythingSubbed && (
          <IconLabelValue
            className={cn('sub-icon')}
            iconSource={'google'}
            icon='swap_horiz'
            tooltipValue='Substitute'
          />
        )}
      </span>
    </div>
  );
}
 
const itemPriceShape = PropTypes.shape({
  amount: PropTypes.number.isRequired,
  currency: PropTypes.string.isRequired,
});
 
LineItemHeader.propTypes = {
  data: PropTypes.shape({
    status: PropTypes.string,
    pickedResult: PropTypes.string,
    orderedLine: PropTypes.shape({
      actualUnitPrice: itemPriceShape.isRequired,
      averageWeight: PropTypes.number.isRequired,
      description: PropTypes.string.isRequired,
      ean: PropTypes.string.isRequired,
      nonPickable: PropTypes.bool.isRequired,
      orderedQuantity: PropTypes.number.isRequired,
      regularUnitPrice: itemPriceShape.isRequired,
      sku: PropTypes.string.isRequired,
      substituteProductAllowed: PropTypes.bool.isRequired,
      unitOfMeasure: PropTypes.string.isRequired,
      weightRange: PropTypes.string,
    }).isRequired,
    pickedLines: PropTypes.array,
  }),
  isInitiallyExpanded: PropTypes.bool,
};
