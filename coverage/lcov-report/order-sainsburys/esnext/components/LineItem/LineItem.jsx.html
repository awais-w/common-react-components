import React, { useState } from 'react';
import PropTypes from 'prop-types';
import bem from '@argos/utils/esnext/bem';
import formatMoney from '@cmc/utils/esnext/formatMoney';
import Card from '@cmc/card/esnext/card';
import LineItemBody from './components/LineItemBody';
 
import './line-item.scss';
 
const cn = bem({ prefix: 'cmc-', block: 'line-item' });
 
const LineItem = ({ data, isInitiallyExpanded = false }) => {
  const {
    pickedResult,
    orderedLine: { sku, description, actualUnitPrice, regularUnitPrice, orderedQuantity, unitOfMeasure },
    pickedLines,
  } = data;
 
  const [isPanelExpanded, setIsPanelExpanded] = useState(isInitiallyExpanded);
  const isEverythingDelivered = pickedLines.every(
    (pickedLine) => !pickedLine.returnReason && !pickedLine.nonDelivered.length,
  );
  const isDifferentPrice = regularUnitPrice.amount !== actualUnitPrice.amount;
  const isAnythingSubbed = pickedResult === 'SUBSTITUTION' || pickedResult === 'PART_SUBSTITUTION';
  const isShortLife = pickedLines.some((x) => x.shortLife);
  const pickedQuantity = pickedLines.reduce((acc, pickedLine) => {
    acc += pickedLine.quantity;
    return acc;
  }, 0);
  const looseItemUnit = unitOfMeasure.toUpperCase() !== 'EACH' ? ` ${unitOfMeasure.toUpperCase()}` : '';
  const quantity = `${orderedQuantity}${looseItemUnit} ${
    pickedQuantity !== orderedQuantity && pickedResult !== 'PERFECT_PICK'
      ? `(${pickedQuantity}${looseItemUnit} picked) `
      : ''
  }`;
  const pricing = isDifferentPrice
    ? `${formatMoney(regularUnitPrice.amount, regularUnitPrice.currency)}\u00A0\u00A0\u00A0`
    : '';
  const perUnit = `${formatMoney(actualUnitPrice.amount, actualUnitPrice.currency)} / ${unitOfMeasure.toUpperCase()}`;
  const nonDeliveredItem = !isEverythingDelivered && { icon: 'warning', tooltip: 'Return', color: 'red' };
  const substitutionIcon = isAnythingSubbed && { icon: 'swap_horiz', tooltip: 'Substitute', className: cn('sub-icon') };
  const shortLifeIcon = isShortLife && { icon: 'timer', tooltip: 'Short Life', className: cn('short-life-icon') };
  const rightIcons = [nonDeliveredItem, substitutionIcon, shortLifeIcon];
 
  return (
    <div className={cn()}>
      <Card
        isExpanded={isPanelExpanded}
        onToggle={() => setIsPanelExpanded(!isPanelExpanded)}
        key={'contact'}
        title1={sku}
        title2={description}
        title3={`${quantity}x ${pricing}Unit price: ${perUnit}`}
        rightIcons={rightIcons}
        expandedPanelType={'panel-info'}
        collapsedPanelType={'panel-info'}
        isNested
        role={'none'}>
        <LineItemBody data={data} />
      </Card>
    </div>
  );
};
 
const itemPriceShape = PropTypes.shape({
  amount: PropTypes.number.isRequired,
  currency: PropTypes.string.isRequired,
});
 
LineItem.propTypes = {
  data: PropTypes.shape({
    status: PropTypes.string,
    pickedResult: PropTypes.string,
    orderedLine: PropTypes.shape({
      actualUnitPrice: itemPriceShape.isRequired,
      averageWeight: PropTypes.number.isRequired,
      description: PropTypes.string.isRequired,
      ean: PropTypes.string.isRequired,
      nonPickable: PropTypes.bool.isRequired,
      orderedQuantity: PropTypes.number.isRequired,
      regularUnitPrice: itemPriceShape.isRequired,
      sku: PropTypes.string.isRequired,
      substituteProductAllowed: PropTypes.bool.isRequired,
      unitOfMeasure: PropTypes.string.isRequired,
      weightRange: PropTypes.string,
    }).isRequired,
    pickedLines: PropTypes.array,
  }),
  isInitiallyExpanded: PropTypes.bool,
};
 
export default LineItem;
