import React from 'react';
import PropTypes from 'prop-types';
import Skeleton from 'react-loading-skeleton';
import bem from '@argos/utils/esnext/bem';
import Pane from '@cmc/pane/esnext/pane';
import IconLabelValue from '@cmc/icon-label-value/esnext/IconLabelValue';
import Notification from '@cmc/cha-notification/esnext/Notification';
import Table from '@cmc/table/esnext/Table';
import isEmpty from '@cmc/utils/esnext/isEmpty';
import Button from '@cmc/cha-button/esnext/Button';
import formatMoney from '@cmc/utils/esnext/formatMoney';
import { voucherHeadings, transactionHeadings } from './heading/PaymentDetailsHeading';
import { generateVoucherData, generateTransactionData } from './helpers/PaymentDetailsFunctions';
 
import PaymentStatus from './components/payment-status';
import './PaymentDetails.scss';
 
const cn = bem({ prefix: 'cmc-', block: 'payment-details' });
 
const PaymentDetails = ({
  colleagueDiscountCardNumber = '',
  paymentVouchers = [],
  totalAmount,
  orderTransactions = [],
  status,
  canRetryPayment,
  userHasRetryPermissions,
  onRetryPayment,
  retryPaymentStatus,
  isLoading,
}) => {
  const voucherTableData = generateVoucherData(paymentVouchers);
  const isVoucherData = voucherTableData.length > 0;
  const transactionTableData = generateTransactionData(orderTransactions);
  const { amount, currency } = totalAmount || {};
 
  const buildPaymentStatus = () => {
    switch (retryPaymentStatus) {
      case 'APPROVED':
        return <PaymentStatus type='success' description='PAYMENT SUCCESSFUL' />;
      case 'FAILURE':
        return <PaymentStatus type='failure' description='PAYMENT FAILURE' />;
      case 'DECLINED':
        return <PaymentStatus type='failure' description='PAYMENT DECLINED' />;
      case 'UNKNOWN':
      case 'CANCELLED':
      case 'PENDING':
        return <PaymentStatus type='other' description={retryPaymentStatus} />;
      default:
        return <PaymentStatus type={status.type} description={status.description} />;
    }
  };
 
  const buildRetryNotification = () => {
    switch (retryPaymentStatus) {
      case 'FAILURE':
        return <Notification mode='error'>An unexpected failure has occurred, please retry payment</Notification>;
      case 'DECLINED':
        return <Notification mode='error'>The payment method has been declined</Notification>;
      case 'CANCELLED':
        return <Notification mode='warn'>The payment method has been cancelled</Notification>;
      default:
        return null;
    }
  };
 
  return (
    <div className={cn()}>
      <Pane
        title1={'Payment'}
        title2={!isLoading && `Total: ${formatMoney(amount, currency)}`}
        icon='payment'
        titleRight={!isLoading && buildPaymentStatus()}>
        {canRetryPayment && userHasRetryPermissions && retryPaymentStatus !== 'APPROVED' && (
          <div className={cn('failed-payment')}>
            {retryPaymentStatus && buildRetryNotification()}
            <div className={cn('failed-payment-buttons')}>
              <Button
                isSecondary
                isShort
                buttonWidth='contentFit'
                text='New Card Payment'
                onClick={() => onRetryPayment(false)}
              />
 
              <Button isShort buttonWidth='contentFit' text='Retry Payment' onClick={() => onRetryPayment(true)} />
            </div>
          </div>
        )}
        <div className={cn('discount-card')}>
          {isLoading ? (
            <Skeleton />
          ) : (
            <IconLabelValue label='Discount card #:' value={colleagueDiscountCardNumber || 'N/A'} />
          )}
        </div>
        <div className={cn('vouchers')}>
          {isLoading ? <Skeleton /> : <IconLabelValue label='Vouchers:' value={isVoucherData ? undefined : 'N/A'} />}
          {isVoucherData && (
            <Table
              className={cn('vouchers-data')}
              headings={voucherHeadings}
              striped
              bordered
              tableData={voucherTableData}
            />
          )}
        </div>
        <div className={cn('transactions')}>
          {isLoading ? (
            <Skeleton />
          ) : (
            <IconLabelValue
              label='Transactions:'
              value={!isEmpty(transactionTableData) ? undefined : 'No transactions'}
            />
          )}
          {!isEmpty(transactionTableData) && (
            <Table
              className={cn('transactions-data')}
              headings={transactionHeadings}
              tableData={transactionTableData}
              striped
              bordered
            />
          )}
        </div>
      </Pane>
    </div>
  );
};
 
PaymentDetails.propTypes = {
  colleagueDiscountCardNumber: PropTypes.string,
  paymentVouchers: PropTypes.arrayOf(
    PropTypes.shape({
      ean: PropTypes.string,
      quantity: PropTypes.number,
    }),
  ),
  isRefunded: PropTypes.bool,
  totalAmount: PropTypes.shape({
    amount: PropTypes.number,
    currency: PropTypes.string,
  }).isRequired,
  orderTransactions: PropTypes.arrayOf(
    PropTypes.shape({
      total: PropTypes.shape({
        amount: PropTypes.number,
        currency: PropTypes.string,
      }),
      timestamp: PropTypes.any,
      orderTransactionType: PropTypes.oneOf(['PAYMENT', 'REFUND']),
    }),
  ).isRequired,
  status: PropTypes.string,
  canRetryPayment: PropTypes.bool,
  userHasRetryPermissions: PropTypes.bool,
  onRetryPayment: PropTypes.func,
  retryPaymentStatus: PropTypes.string,
};
export default PaymentDetails;
