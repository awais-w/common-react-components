import React from 'react';
import PropTypes from 'prop-types';
import bem from '@argos/utils/esnext/bem';
import './PaymentStatus.scss';
import IconLabelValue from '@cmc/icon-label-value/esnext/IconLabelValue';
 
const cn = bem({ prefix: 'cmc-', block: 'payment-status' });
 
const PaymentStatus = ({ type, description }) => (
  <div className={cn()}>
    <span className={cn('label')}>Status</span>
    <div className={cn('status', type)}>
      <IconLabelValue
        icon={type === 'success' ? 'check_circle' : type === 'failure' ? 'cancel' : ''}
        label={description || 'UNKNOWN'}
      />
    </div>
  </div>
);
 
PaymentStatus.propTypes = {
  type: PropTypes.oneOf(['success', 'failure', 'other']),
  description: PropTypes.string,
};
 
export default PaymentStatus;
