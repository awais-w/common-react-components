const defaultProps = {

  alwaysShowPaginationButtons: false,

class PaginationMatrix extends Component {
  constructor(props) {
    super(props);
    this.state = {
      visibleColumns: this.props.visibleColumns,
      columnOffset: 0,
      columnsShouldIncrementBy: this.props.visibleColumns,
      activeSetting: {
        rowId: null,
        cellId: null,
      },
    };
 
    this.handleShouldPageUpdate = this.handleShouldPageUpdate.bind(this);
    this.handleNextPage = this.handleNextPage.bind(this);
    this.handleCellClick = this.handleCellClick.bind(this);
  }
 
  componentWillReceiveProps(nextProps) {
    if (nextProps.matrixData.headings.length !== this.props.matrixData.headings.length) {
      this.handleShouldPageUpdate({ desiredDirection: 'next' });
    }
  }
 
  handleShouldPageUpdate({ desiredDirection }) {
    const {
      onNextBatchRequest,
      matrixData: { headings },
    } = this.props;
    const { columnOffset, columnsShouldIncrementBy, visibleColumns } = this.state;
 
    /**
     * Based on the direction, we either add or subtract the incrementation number from the offset
     * This redraws the grid with new positional numbers
     */
    const directionalOffset = desiredDirection === 'prev' ? -columnsShouldIncrementBy : columnsShouldIncrementBy;
    const newOffset = columnOffset + directionalOffset;
    const newVisibleColumns = visibleColumns + directionalOffset;
 
    /**
     * After we've calculated the new positions above, we need check if there is data for those positions
     * We look to see if there is a heading in the heading array in the new position
     * If there is, we redraw the table with that data
     * Otherwise we will call onNextBatchRequest to get more data
     */
    if (headings[newOffset]) {
      this.handleNextPage(newOffset, newVisibleColumns);
    } else {
      onNextBatchRequest();
    }
  }
 
  handleNextPage(offset, cellsToDisplay) {
    this.setState(() => ({
      columnOffset: offset,
      visibleColumns: cellsToDisplay,
    }));
  }
 
  handleCellClick(cellIndex, rowIndex) {
    this.props.onClickCell({ cellIndex, rowIndex });
 
    this.setState(() => ({
      activeSetting: {
        rowId: rowIndex,
        cellId: cellIndex,
      },
    }));
  }
 
  render() {
    const {
      matrixData: { headings, rows, sideHeadings },
      onNextBatchRequest,
      isLoading,
      alwaysShowPaginationButtons,
    } = this.props;
    const { activeSetting, visibleColumns, columnOffset, columnsShouldIncrementBy } = this.state;
    // cell size is calculated based on how many cells should be shown
    const cellSizeCalc = 100 / (this.props.visibleColumns + 1);
 
    const matrixHeadings =
      !isEmpty(headings) &&
      headings.map(
        (heading, headingIndex) =>
          // we work out which cells should and shouldn't show based on internal calculations
          // if the index is not with columnOffset and visibleColumns, we don't show it
          headingIndex >= columnOffset &&
          headingIndex < visibleColumns && (
            <div
              className={`PaginationMatrix--heading
            ${activeSetting.cellId === headingIndex ? 'PaginationMatrix--heading-active' : ''}`}
              style={{ width: `${cellSizeCalc}%` }}
              key={`heading-${headingIndex}`}
              id={`heading-${headingIndex}`}>
              {heading}
            </div>
          ),
      );
 
    const matrixRows =
      rows &&
      rows.map((row, rowIndex) => {
        const activeRow = activeSetting.rowId === rowIndex;
 
        const rowStart = sideHeadings ? (
          <div
            className={`PaginationMatrix--sideHeading ${activeRow ? 'PaginationMatrix--sideHeading-active' : ''}`}
            style={{ width: `${cellSizeCalc}%` }}
            key={`sideheading-${rowIndex}`}
            id={`sideheading-${rowIndex}`}>
            {sideHeadings[rowIndex]}
          </div>
        ) : (
          ''
        );
 
        const cells = row.map(
          (cell, cellIndex) =>
            // we again look to see if the cell should be shown, based on it's position, as above with the headings
            cellIndex >= columnOffset &&
            cellIndex < visibleColumns && (
              <div
                className={`PaginationMatrix--cell
              ${activeRow && activeSetting.cellId === cellIndex ? 'PaginationMatrix--cell-active' : ''}
              ${cell === 'disabled' ? 'PaginationMatrix--disabled' : ''}`}
                style={{ width: `${cellSizeCalc}%` }}
                onClick={cell !== 'disabled' ? () => this.handleCellClick(cellIndex, rowIndex) : undefined}
                key={`row-${rowIndex}-cell-${cellIndex}`}
                id={`row-${rowIndex}-cell-${cellIndex}`}>
                {cell !== 'disabled' && cell}
              </div>
            ),
        );
 
        return (
          <div className='PaginationMatrix--row' key={`row-${rowIndex}`} id={`row-${rowIndex}`}>
            {rowStart}
            {cells}
          </div>
        );
      });
 
    // if headings are empty we still need to get a length.
    // We check the first row because if there are rows, the first one should be populated
    const incrementCheck = headings ? headings.length : rows && rows[0].length;
    const shouldTableDecrement = visibleColumns - columnsShouldIncrementBy >= this.props.visibleColumns;
    const shouldTableIncrement = onNextBatchRequest || incrementCheck > visibleColumns; // onNextBatchRequest !== null || incrementCheck > visibleColumns;
 
    return (
      <div className='PaginationMatrix bolt-grids'>
        <div className='xs-row'>
          <div className='PaginationMatrix--arrowContainer xs-1'>
            <div
              className={`PaginationMatrix--arrow PaginationMatrix--prev ${
                !shouldTableDecrement ? 'PaginationMatrix--disabled' : ''
              }`}>
              {(this.props.alwaysShowPaginationButtons || shouldTableDecrement) && (
                <Icon
                  icon='keyboard_arrow_left'
                  onClick={() => {
                    this.handleShouldPageUpdate({ desiredDirection: 'prev' });
                  }}
                />
              )}
            </div>
          </div>
          <div className='PaginationMatrix--table xs-10'>
            {!isEmpty(headings) && (
              <div className='PaginationMatrix--row'>
                {!isEmpty(sideHeadings) && (
                  <div className='PaginationMatrix--blank' style={{ width: `${cellSizeCalc}%` }} />
                )}
                {matrixHeadings}
              </div>
            )}
            {isLoading ? (
              <div className='PaginationMatrix--row'>
                <LoadingWheel contained={true} />
              </div>
            ) : (
              matrixRows
            )}
          </div>
 
          <div className='PaginationMatrix--arrowContainer xs-1'>
            <div
              className={`PaginationMatrix--arrow PaginationMatrix--next ${
                !shouldTableIncrement ? 'PaginationMatrix--disabled' : ''
              }`}>
              {(alwaysShowPaginationButtons || shouldTableIncrement || onNextBatchRequest) && (
                <Icon
                  icon='keyboard_arrow_right'
                  onClick={() => {
                    this.handleShouldPageUpdate({ desiredDirection: 'next' });
                  }}
                />
              )}
            </div>
          </div>
        </div>
      </div>
    );
  }
}
