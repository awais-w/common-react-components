import React from 'react';
import PropTypes from 'prop-types';
import bem from '@argos/utils/esnext/bem';
import './checkbox.scss';
 
const icons = ['tick', 'cross'];
 
const propTypes = {
  id: PropTypes.string.isRequired,
  children: PropTypes.any,
  checked: PropTypes.bool.isRequired,
  onChange: PropTypes.func.isRequired,
  icon: PropTypes.oneOf(icons),
  isDisabled: PropTypes.bool,
};
 
const defaultProps = {
  icon: 'tick',
  isDisabled: false,
};
 
const cn = bem({ prefix: 'cmc-', block: 'checkbox' });
 
const Checkbox = (props) => {
  const { id, children, onChange, checked, icon, isDisabled } = props;
  return (
    <div className={cn()}>
      <input
        id={id}
        className={cn('input', icon)}
        disabled={isDisabled}
        type='checkbox'
        onChange={onChange}
        checked={checked}
      />
      <label htmlFor={id} className={cn('label')}>
        {children}
      </label>
    </div>
  );
};
 
Checkbox.propTypes = propTypes;
Checkbox.defaultProps = defaultProps;
export default Checkbox;
